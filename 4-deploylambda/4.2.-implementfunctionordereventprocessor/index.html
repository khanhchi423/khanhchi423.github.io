Implement OrderEventProcessor

Next, we will create the OrderEventProcessor Lambda function, which listens to order events and updates the OrderReadModel accordingly.

The steps to create this function are similar to those in the previous section.

Step 1: Create Lambda Function

  1. Select Author from scratch.
  2. Enter the function name: OrderEventProcessor.
  3. Choose Runtime: Python 3.12.
  4. Choose Architecture: x86_64.
  5. select Use an existing role and pick EventSourcingLambdaRole.
  6. Click Create function.

Step 2: Implement Function Code

  1. Replace the default code with the following:
import json
import boto3
from decimal import Decimal
from datetime import datetime

# Initialize AWS clients
dynamodb = boto3.resource('dynamodb')
sns = boto3.client('sns')

# DynamoDB tables
order_table = dynamodb.Table('OrderReadModel')
customer_table = dynamodb.Table('CustomerReadModel')

# SNS Topic ARN - Replace with your actual SNS topic ARN
SNS_TOPIC_ARN = 'arn:aws:sns:ap-southeast-1:218451864254:OrderNotifications'

def lambda_handler(event, context):
    try:
        print(f"Received event: {json.dumps(event)}")
        
        # Handle EventBridge events
        if 'detail-type' in event and 'detail' in event:
            process_eventbridge_event(event)
        # Handle direct invocation or other event sources
        elif 'Records' in event:
            for record in event['Records']:
                if 'eventbridge' in record.get('eventSource', ''):
                    process_eventbridge_event(json.loads(record['body']))
        else:
            # Direct EventBridge invocation
            process_eventbridge_event(event)
        
        return {'statusCode': 200, 'body': 'Events processed successfully'}
        
    except Exception as e:
        print(f"Error processing events: {str(e)}")
        import traceback
        traceback.print_exc()
        return {'statusCode': 500, 'body': 'Error processing events'}

def process_eventbridge_event(event_data):
    detail_type = event_data.get('detail-type')
    detail = event_data.get('detail', {})
    
    print(f"Processing event type: {detail_type}")
    print(f"Event detail: {detail}")
    
    if detail_type == 'OrderPlaced':
        handle_order_placed(detail)
    elif detail_type == 'OrderConfirmed':
        handle_order_confirmed(detail)
    elif detail_type == 'OrderCancelled':
        handle_order_cancelled(detail)
    else:
        print(f"Unknown event type: {detail_type}")

def handle_order_placed(detail):
    try:
        # Convert float values back to Decimal for DynamoDB
        items = []
        for item in detail.get('items', []):
            items.append({
                'itemId': item['itemId'],
                'name': item['name'],
                'price': Decimal(str(item['price'])),
                'quantity': item['quantity'],
                'total': Decimal(str(item['total']))
            })

        # Create order record in read model
        order_item = {
            'orderId': detail['orderId'],
            'customerName': detail['customerName'],
            'customerPhone': detail['customerPhone'],
            'customerAddress': detail['customerAddress'],
            'items': items,
            'totalAmount': Decimal(str(detail['totalAmount'])),
            'status': 'PLACED',
            'createdAt': detail.get('timestamp', datetime.utcnow().isoformat()),
            'updatedAt': detail.get('timestamp', datetime.utcnow().isoformat())
        }

        order_table.put_item(Item=order_item)
        print(f"Order {detail['orderId']} created in read model")

        # Update customer record
        customer_id = detail['customerPhone']  # Use phone as customer ID
        try:
            customer_table.update_item(
                Key={'customerId': customer_id},
                UpdateExpression='SET customerName = :name, customerPhone = :phone, totalOrders = if_not_exists(totalOrders, :zero) + :one, updatedAt = :updated',
                ExpressionAttributeValues={
                    ':name': detail['customerName'],
                    ':phone': detail['customerPhone'],
                    ':zero': 0,
                    ':one': 1,
                    ':updated': datetime.utcnow().isoformat()
                }
            )
            print(f"Customer {customer_id} updated")
        except Exception as e:
            print(f"Error updating customer: {str(e)}")

        # Gửi thông báo cho admin
        send_notification(
            f"[ADMIN] New Order Placed - #{detail['orderId'][:8]}",
            (
                f"A new customer order has been placed.\n"
                f"Order ID: {detail['orderId']}\n"
                f"Customer: {detail['customerName']}\n"
                f"Phone: {detail['customerPhone']}\n"
                f"Total: ${detail['totalAmount']:.2f}\n"
                f"Status: PLACED"
            )
        )

    except Exception as e:
        print(f"Error handling OrderPlaced: {str(e)}")
        import traceback
        traceback.print_exc()
        raise


def handle_order_confirmed(detail):
    try:
        order_table.update_item(
            Key={'orderId': detail['orderId']},
            UpdateExpression='SET #status = :status, confirmedAt = :confirmed_at, estimatedDelivery = :estimated, updatedAt = :updated',
            ExpressionAttributeNames={'#status': 'status'},
            ExpressionAttributeValues={
                ':status': 'CONFIRMED',
                ':confirmed_at': detail.get('confirmedAt'),
                ':estimated': detail.get('estimatedDelivery'),
                ':updated': datetime.utcnow().isoformat()
            }
        )
        print(f"Order {detail['orderId']} confirmed")

        # Gửi thông báo cho admin
        send_notification(
            f"[ADMIN] Order Confirmed - #{detail['orderId'][:8]}",
            (
                f"Customer order has been confirmed.\n"
                f"Order ID: {detail['orderId']}\n"
                f"Estimated Delivery: {detail.get('estimatedDelivery', 'TBD')}"
            )
        )

    except Exception as e:
        print(f"Error handling OrderConfirmed: {str(e)}")
        import traceback
        traceback.print_exc()
        raise

def handle_order_cancelled(detail):
    try:
        order_table.update_item(
            Key={'orderId': detail['orderId']},
            UpdateExpression='SET #status = :status, cancelledAt = :cancelled_at, cancelReason = :reason, updatedAt = :updated',
            ExpressionAttributeNames={'#status': 'status'},
            ExpressionAttributeValues={
                ':status': 'CANCELLED',
                ':cancelled_at': detail.get('cancelledAt'),
                ':reason': detail.get('reason'),
                ':updated': datetime.utcnow().isoformat()
            }
        )
        print(f"Order {detail['orderId']} cancelled")

        # Gửi thông báo cho admin
        send_notification(
            f"[ADMIN] Order Cancelled - #{detail['orderId'][:8]}",
            (
                f"Customer order has been cancelled.\n"
                f"Order ID: {detail['orderId']}\n"
                f"Reason: {detail.get('reason', 'No reason provided')}"
            )
        )

    except Exception as e:
        print(f"Error handling OrderCancelled: {str(e)}")
        import traceback
        traceback.print_exc()
        raise

def send_notification(subject, message):
    try:
        print(f"NOTIFICATION - {subject}: {message}")
        sns.publish(
            TopicArn=SNS_TOPIC_ARN,
            Subject=str(subject),
            Message=str(message)
        )
    except Exception as e:
        print(f"Error sending notification: {str(e)}")
        import traceback
        traceback.print_exc()
        # Không raise để Lambda tiếp tục chạy
