Implement State Reconstruction Handler

Finally, we will implement the State Reconstruction Handler Lambda function. This function rebuilds the current state of an order by replaying events from the Event Store, ensuring our system’s state is consistent with the sequence of events.

The creation steps are similar to the previous Lambda functions.

Step 1: Create Lambda Function

  1. Select Author from scratch.
  2. Enter the function name: StateReconstructionHandler.
  3. Choose Runtime: Python 3.12.
  4. Choose Architecture: x86_64.
  5. Select Use an existing role and pick EventSourcingLambdaRole.
  6. Click Create function.

Step 2: Implement Function Code

  1. Replace the default code with the following:
import json
import boto3
from boto3.dynamodb.conditions import Key
from decimal import Decimal
from datetime import datetime

# Initialize DynamoDB
dynamodb = boto3.resource('dynamodb')
event_store_table = dynamodb.Table('EventStore')
order_table = dynamodb.Table('OrderReadModel')

class DecimalEncoder(json.JSONEncoder):
    def default(self, o):
        if isinstance(o, Decimal):
            return float(o)
        return super(DecimalEncoder, self).default(o)

def lambda_handler(event, context):
    """
    State Reconstruction Handler
    - HTTP request: rebuild theo API path
    - EventBridge event: rebuild ngay cho order liên quan
    """
    try:
        print("Received event:", json.dumps(event))

        http_method = None
        path = None
        body = {}

        # Trường hợp API Gateway REST API
        if 'httpMethod' in event:
            http_method = event['httpMethod']
            path = event.get('path')
            body = json.loads(event.get('body') or "{}")

        # Trường hợp API Gateway HTTP API (v2 payload)
        elif event.get('requestContext', {}).get('http', {}).get('method'):
            http_method = event['requestContext']['http']['method']
            path = event.get('rawPath')
            body = json.loads(event.get('body') or "{}")

        # Trường hợp EventBridge event
        elif event.get("detail-type") in ["OrderPlaced", "OrderConfirmed", "OrderCancelled"]:
            order_id = event.get("detail", {}).get("orderId")
            if not order_id:
                print("No orderId in event, skipping reconstruction.")
                return build_response(400, {"error": "orderId not found in event"})

            print(f"Rebuilding state for orderId from EventBridge: {order_id}")
            events = get_events_for_order(order_id)
            if not events:
                return build_response(404, {"error": f"No events found for order {order_id}"})
            reconstructed_state = reconstruct_order_state(events)
            update_read_model(reconstructed_state)

            return build_response(200, {
                "message": f"State rebuilt from EventBridge event for order {order_id}",
                "orderId": order_id,
                "eventsProcessed": len(events),
                "reconstructedState": reconstructed_state
            })

        else:
            print("Unknown event source, skipping.")
            return build_response(200, {"message": "Event ignored"})

        # Xử lý HTTP request như cũ
        print(f"Processing {http_method} {path}")
        if path == '/reconstruct/order' and http_method == 'POST':
            return handle_reconstruct_order(body)
        elif path == '/reconstruct/all' and http_method == 'POST':
            return handle_reconstruct_all_orders(body)
        elif path == '/reconstruct/validate' and http_method == 'POST':
            return handle_validate_reconstruction(body)
        else:
            return build_response(404, {'error': 'Endpoint not found'})

    except Exception as e:
        print(f"Error: {str(e)}")
        import traceback
        traceback.print_exc()
        return build_response(500, {'error': 'Internal server error'})


def handle_reconstruct_order(body):
    """Reconstruct a single order from events"""
    order_id = body.get('orderId')
    if not order_id:
        return build_response(400, {'error': 'orderId is required'})
    
    try:
        # Get all events for this order
        events = get_events_for_order(order_id)
        
        if not events:
            return build_response(404, {'error': 'No events found for order'})
        
        # Reconstruct state from events
        reconstructed_state = reconstruct_order_state(events)
        
        # Update read model
        update_read_model(reconstructed_state)
        
        return build_response(200, {
            'message': 'Order state reconstructed successfully',
            'orderId': order_id,
            'eventsProcessed': len(events),
            'reconstructedState': reconstructed_state
        })
        
    except Exception as e:
        print(f"Error reconstructing order {order_id}: {str(e)}")
        return build_response(500, {'error': 'Failed to reconstruct order'})

def handle_reconstruct_all_orders(body):
    """Reconstruct all orders from events"""
    max_orders = body.get('maxOrders', 100)  # Limit for safety
    
    try:
        # Get all unique aggregate IDs from event store
        response = event_store_table.scan(
            ProjectionExpression='aggregateId',
            Limit=max_orders
        )
        
        order_ids = list(set([item['aggregateId'] for item in response.get('Items', [])]))
        
        reconstructed_orders = []
        errors = []
        
        for order_id in order_ids:
            try:
                events = get_events_for_order(order_id)
                if events:
                    reconstructed_state = reconstruct_order_state(events)
                    update_read_model(reconstructed_state)
                    reconstructed_orders.append({
                        'orderId': order_id,
                        'eventsProcessed': len(events)
                    })
            except Exception as e:
                errors.append({
                    'orderId': order_id,
                    'error': str(e)
                })
        
        return build_response(200, {
            'message': f'Reconstructed {len(reconstructed_orders)} orders',
            'reconstructedOrders': reconstructed_orders,
            'errors': errors
        })
        
    except Exception as e:
        print(f"Error reconstructing all orders: {str(e)}")
        return build_response(500, {'error': 'Failed to reconstruct orders'})

def handle_validate_reconstruction(body):
    """Validate reconstruction by comparing with current read model"""
    order_id = body.get('orderId')
    if not order_id:
        return build_response(400, {'error': 'orderId is required'})
    
    try:
        # Get current read model state
        current_state = order_table.get_item(Key={'orderId': order_id}).get('Item')
        
        # Get reconstructed state from events
        events = get_events_for_order(order_id)
        reconstructed_state = reconstruct_order_state(events)
        
        # Compare states
        validation_result = compare_states(current_state, reconstructed_state)
        
        return build_response(200, {
            'orderId': order_id,
            'isValid': validation_result['isValid'],
            'differences': validation_result['differences'],
            'currentState': current_state,
            'reconstructedState': reconstructed_state
        })
        
    except Exception as e:
        print(f"Error validating reconstruction: {str(e)}")
        return build_response(500, {'error': 'Failed to validate reconstruction'})

def get_events_for_order(order_id):
    """Get all events for a specific order, sorted by timestamp"""
    try:
        response = event_store_table.query(
            IndexName='AggregateIndex',  # Assuming GSI on aggregateId
            KeyConditionExpression=Key('aggregateId').eq(order_id),
            ScanIndexForward=True  # Sort by timestamp ascending
        )
        
        events = response.get('Items', [])
        
        # Sort by timestamp to ensure correct order
        events.sort(key=lambda x: x.get('timestamp', ''))
        
        return events
        
    except Exception as e:
        print(f"Error getting events for order {order_id}: {str(e)}")
        # Fallback to scan if GSI doesn't exist
        response = event_store_table.scan(
            FilterExpression=Key('aggregateId').eq(order_id)
        )
        events = response.get('Items', [])
        events.sort(key=lambda x: x.get('timestamp', ''))
        return events

def reconstruct_order_state(events):
    """Reconstruct order state by replaying events"""
    state = {
        'orderId': None,
        'status': None,
        'customerName': None,
        'customerPhone': None,
        'customerAddress': None,
        'items': [],
        'totalAmount': Decimal('0'),
        'createdAt': None,
        'updatedAt': None,
        'confirmedAt': None,
        'cancelledAt': None,
        'cancelReason': None,
        'estimatedDelivery': None,
        'eventHistory': []
    }
    
    for event in events:
        event_type = event.get('eventType')
        event_data = event.get('eventData', {})
        timestamp = event.get('timestamp')
        
        # Track event history
        state['eventHistory'].append({
            'eventType': event_type,
            'timestamp': timestamp,
            'eventId': event.get('eventId')
        })
        
        if event_type == 'OrderPlaced':
            state['orderId'] = event_data.get('orderId')
            state['customerName'] = event_data.get('customerName')
            state['customerPhone'] = event_data.get('customerPhone')
            state['customerAddress'] = event_data.get('customerAddress')
            state['items'] = event_data.get('items', [])
            state['totalAmount'] = Decimal(str(event_data.get('totalAmount', 0)))
            state['status'] = 'PLACED'
            state['createdAt'] = timestamp
            state['updatedAt'] = timestamp
            
        elif event_type == 'OrderConfirmed':
            state['status'] = 'CONFIRMED'
            state['confirmedAt'] = event_data.get('confirmedAt')
            state['estimatedDelivery'] = event_data.get('estimatedDelivery')
            state['updatedAt'] = timestamp
            
        elif event_type == 'OrderCancelled':
            state['status'] = 'CANCELLED'
            state['cancelledAt'] = event_data.get('cancelledAt')
            state['cancelReason'] = event_data.get('reason')
            state['updatedAt'] = timestamp
            
        # Add more event types as needed
        
    return state

def update_read_model(state):
    """Update the read model with reconstructed state"""
    try:
        # Remove event history for read model
        read_model_state = {k: v for k, v in state.items() if k != 'eventHistory'}
        
        order_table.put_item(Item=read_model_state)
        print(f"Updated read model for order {state['orderId']}")
        
    except Exception as e:
        print(f"Error updating read model: {str(e)}")
        raise

def compare_states(current_state, reconstructed_state):
    """Compare current read model state with reconstructed state"""
    differences = []
    
    if not current_state:
        return {
            'isValid': False,
            'differences': ['Current state not found in read model']
        }
    
    # Remove event history from comparison
    reconstructed_compare = {k: v for k, v in reconstructed_state.items() if k != 'eventHistory'}
    
    # Compare key fields
    compare_fields = ['orderId', 'status', 'customerName', 'customerPhone', 
                     'totalAmount', 'createdAt', 'confirmedAt', 'cancelledAt']
    
    for field in compare_fields:
        current_value = current_state.get(field)
        reconstructed_value = reconstructed_compare.get(field)
        
        # Handle Decimal comparison
        if isinstance(current_value, Decimal) and isinstance(reconstructed_value, Decimal):
            if current_value != reconstructed_value:
                differences.append(f"{field}: current={current_value}, reconstructed={reconstructed_value}")
        elif str(current_value) != str(reconstructed_value):
            differences.append(f"{field}: current={current_value}, reconstructed={reconstructed_value}")
    
    return {
        'isValid': len(differences) == 0,
        'differences': differences
    }

def build_response(status_code, body):
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type, Authorization'
        },
        'body': json.dumps(body, cls=DecimalEncoder, default=str)
    }
