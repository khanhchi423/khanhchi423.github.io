Triển khai OrderQueryHandler

Tiếp theo, chúng ta sẽ tạo hàm Lambda OrderQueryHandler để xử lý các yêu cầu truy vấn và lấy thông tin chi tiết đơn hàng từ OrderReadModel.

Các bước tạo hàm này tương tự như những phần trước.

Bước 1: Tạo hàm Lambda

  1. Chọn Author from scratch.
  2. Nhập tên hàm: OrderQueryHandler.
  3. Chọn Runtime: Python 3.12.
  4. Chọn Architecture: x86_64.
  5. Chọn Use an existing role và chọn EventSourcingLambdaRole.
  6. Nhấn Create function.

Bước 2: Triển khai mã hàm

  1. Thay thế mã mặc định bằng đoạn mã sau:
import json
import boto3
from boto3.dynamodb.conditions import Key
from decimal import Decimal

# Initialize DynamoDB
dynamodb = boto3.resource('dynamodb')
order_table = dynamodb.Table('OrderReadModel')
customer_table = dynamodb.Table('CustomerReadModel')
menu_items_table = dynamodb.Table('MenuItems')

class DecimalEncoder(json.JSONEncoder):
    def default(self, o):
        if isinstance(o, Decimal):
            return float(o)
        return super(DecimalEncoder, self).default(o)

def lambda_handler(event, context):
    try:
        http_method = event['httpMethod']
        path = event['path']
        query_params = event.get('queryStringParameters') or {}
        
        print(f"Processing {http_method} {path}")
        print(f"Query params: {query_params}")
        
        if path == '/orders' and http_method == 'GET':
            return handle_get_orders(query_params)
        elif path.startswith('/orders/') and http_method == 'GET':
            order_id = path.split('/')[-1]
            return handle_get_order(order_id)
        elif path == '/menu' and http_method == 'GET':
            return handle_get_menu()
        elif path == '/customers' and http_method == 'GET':
            return handle_get_customers(query_params)
        else:
            return build_response(404, {'error': 'Endpoint not found'})
            
    except Exception as e:
        print(f"Error: {str(e)}")
        import traceback
        traceback.print_exc()
        return build_response(500, {'error': 'Internal server error'})

def handle_get_orders(query_params):
    try:
        # Get all orders (in production, you'd want pagination)
        response = order_table.scan(
            Limit=50  # Limit to 50 orders
        )
        
        orders = response.get('Items', [])
        
        # Sort by createdAt descending
        orders.sort(key=lambda x: x.get('createdAt', ''), reverse=True)
        
        return build_response(200, {
            'orders': orders,
            'count': len(orders)
        })
        
    except Exception as e:
        print(f"Error getting orders: {str(e)}")
        return build_response(500, {'error': 'Failed to get orders'})

def handle_get_order(order_id):
    try:
        response = order_table.get_item(Key={'orderId': order_id})
        
        if 'Item' in response:
            return build_response(200, response['Item'])
        else:
            return build_response(404, {'error': 'Order not found'})
            
    except Exception as e:
        print(f"Error getting order: {str(e)}")
        return build_response(500, {'error': 'Failed to get order'})

def handle_get_menu():
    try:
        response = menu_items_table.scan()
        items = response.get('Items', [])
        
        # Filter only available items
        available_items = [item for item in items if item.get('available', False)]
        
        return build_response(200, {
            'items': available_items,
            'count': len(available_items)
        })
        
    except Exception as e:
        print(f"Error getting menu: {str(e)}")
        return build_response(500, {'error': 'Failed to get menu'})

def handle_get_customers(query_params):
    try:
        response = customer_table.scan(
            Limit=50  # Limit to 50 customers
        )
        
        customers = response.get('Items', [])
        
        return build_response(200, {
            'customers': customers,
            'count': len(customers)
        })
        
    except Exception as e:
        print(f"Error getting customers: {str(e)}")
        return build_response(500, {'error': 'Failed to get customers'})

def build_response(status_code, body):
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type, Authorization'
        },
        'body': json.dumps(body, cls=DecimalEncoder, default=str)
    }
