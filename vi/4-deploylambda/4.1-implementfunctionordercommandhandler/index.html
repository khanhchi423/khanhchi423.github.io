Triển khai OrderCommandHandler

Trong phần này, chúng ta sẽ triển khai hàm Lambda OrderCommandHandler để xử lý các lệnh (command) liên quan đến đơn hàng trong hệ thống đặt đồ ăn nhanh.
Hàm này sẽ thực hiện các tác vụ như đặt hàng, xác nhận đơn hàng và hủy đơn hàng.

Bước 1: Mở Lambda Console

  1. Trong AWS Management Console, tìm kiếm Lambda.
  2. Nhấn vào Lambda để mở dịch vụ.

Bước 2: Tạo hàm Lambda

  1. Nhấn Create function.

  2. Chọn Author from scratch.

  3. Nhập Function name: OrderCommandHandler.

  4. Chọn Runtime: Python 3.12.

  5. Chọn Architecture: x86_64.

  6. Mở rộng mục Change default execution role.

  7. Chọn Use an existing role.

  8. Chọn EventSourcingLambdaRole từ danh sách.

  9. Nhấn Create function.

Bước 3: Triển khai mã hàm

  1. Trong phần Code source của hàm OrderCommandHandler, thay thế mã mặc định bằng đoạn sau:
import json
import boto3
import uuid
from datetime import datetime, timedelta
from decimal import Decimal

# Initialize AWS clients
dynamodb = boto3.resource('dynamodb')
eventbridge = boto3.client('events')

# DynamoDB tables
event_store_table = dynamodb.Table('EventStore')
menu_items_table = dynamodb.Table('MenuItems')

def lambda_handler(event, context):
    try:
        # Parse request
        http_method = event['httpMethod']
        path = event['path']
        body = json.loads(event['body']) if event.get('body') else {}
        
        print(f"Processing {http_method} {path}")
        print(f"Body: {body}")
        
        if path == '/orders' and http_method == 'POST':
            return handle_place_order(body)
        elif path == '/orders/confirm' and http_method == 'POST':
            return handle_confirm_order(body)
        elif path == '/orders/cancel' and http_method == 'POST':
            return handle_cancel_order(body)
        else:
            return build_response(404, {'error': 'Endpoint not found'})
            
    except Exception as e:
        print(f"Error: {str(e)}")
        import traceback
        traceback.print_exc()
        return build_response(500, {'error': 'Internal server error'})

def handle_place_order(order_data):
    # Validate order data
    required_fields = ['customerName', 'customerPhone', 'customerAddress', 'items']
    if not all(field in order_data for field in required_fields):
        return build_response(400, {'error': 'Missing required fields'})
    
    # Validate items
    if not order_data['items'] or len(order_data['items']) == 0:
        return build_response(400, {'error': 'Order must contain at least one item'})
    
    # Generate order ID
    order_id = str(uuid.uuid4())
    
    # Calculate total amount
    total_amount = Decimal('0')
    validated_items = []
    
    for item in order_data['items']:
        if 'itemId' not in item or 'quantity' not in item:
            return build_response(400, {'error': 'Each item must have itemId and quantity'})
        
        # Get item price from menu
        try:
            menu_item = menu_items_table.get_item(Key={'itemId': item['itemId']})
            if 'Item' not in menu_item:
                return build_response(400, {'error': f'Item {item["itemId"]} not found'})
            
            menu_item_data = menu_item['Item']
            if not menu_item_data.get('available', False):
                return build_response(400, {'error': f'Item {item["itemId"]} is not available'})
            
            item_price = Decimal(str(menu_item_data['price']))
            quantity = int(item['quantity'])
            item_total = item_price * quantity
            total_amount += item_total
            
            validated_items.append({
                'itemId': item['itemId'],
                'name': menu_item_data['name'],
                'price': item_price,  # Keep as Decimal for DynamoDB
                'quantity': quantity,
                'image': menu_item_data.get('image', ''),
                'total': item_total   # Keep as Decimal for DynamoDB
            })
            
        except Exception as e:
            print(f"Error processing item {item['itemId']}: {str(e)}")
            return build_response(500, {'error': f'Error processing item {item["itemId"]}'})
    
    # Create OrderPlaced event
    event_id = str(uuid.uuid4())
    timestamp = datetime.utcnow().isoformat()
    
    event_data = {
        'eventId': event_id,
        'aggregateId': order_id,
        'eventType': 'OrderPlaced',
        'eventData': {
            'orderId': order_id,
            'customerName': order_data['customerName'],
            'customerPhone': order_data['customerPhone'],
            'customerAddress': order_data['customerAddress'],
            'items': validated_items,
            'totalAmount': total_amount,  # Keep as Decimal for DynamoDB
            'status': 'PLACED',
            'timestamp': timestamp
        },
        'timestamp': timestamp,
        'version': 1
    }
    
    try:
        # Store event in Event Store
        event_store_table.put_item(Item=event_data)
        print(f"Event stored successfully in EventStore")
        
        # Publish event to EventBridge (convert Decimals to float for JSON serialization)
        publish_event_to_bridge(event_data)
        print(f"Event published to EventBridge successfully")
        
        return build_response(201, {
            'orderId': order_id,
            'message': 'Order placed successfully',
            'totalAmount': float(total_amount),  # Convert to float for JSON response
            'items': convert_decimals_to_float(validated_items),  # Convert for JSON response
            'status': 'PLACED'
        })
        
    except Exception as e:
        print(f"Error storing event: {str(e)}")
        import traceback
        traceback.print_exc()
        return build_response(500, {'error': 'Failed to process order'})

def handle_confirm_order(confirm_data):
    order_id = confirm_data.get('orderId')
    if not order_id:
        return build_response(400, {'error': 'Order ID required'})
    
    # Calculate estimated delivery time (30 minutes from now)
    estimated_delivery = (datetime.utcnow() + timedelta(minutes=30)).isoformat()
    
    # Create OrderConfirmed event
    event_id = str(uuid.uuid4())
    timestamp = datetime.utcnow().isoformat()
    
    event_data = {
        'eventId': event_id,
        'aggregateId': order_id,
        'eventType': 'OrderConfirmed',
        'eventData': {
            'orderId': order_id,
            'confirmedAt': timestamp,
            'estimatedDelivery': estimated_delivery,
            'status': 'CONFIRMED'
        },
        'timestamp': timestamp,
        'version': 1
    }
    
    try:
        # Store event
        event_store_table.put_item(Item=event_data)
        
        # Publish event
        publish_event_to_bridge(event_data)
        
        return build_response(200, {
            'message': 'Order confirmed successfully',
            'estimatedDelivery': estimated_delivery
        })
        
    except Exception as e:
        print(f"Error confirming order: {str(e)}")
        return build_response(500, {'error': 'Failed to confirm order'})

def handle_cancel_order(cancel_data):
    order_id = cancel_data.get('orderId')
    reason = cancel_data.get('reason', 'Customer request')
    
    if not order_id:
        return build_response(400, {'error': 'Order ID required'})
    
    # Create OrderCancelled event
    event_id = str(uuid.uuid4())
    timestamp = datetime.utcnow().isoformat()
    
    event_data = {
        'eventId': event_id,
        'aggregateId': order_id,
        'eventType': 'OrderCancelled',
        'eventData': {
            'orderId': order_id,
            'cancelledAt': timestamp,
            'reason': reason,
            'status': 'CANCELLED'
        },
        'timestamp': timestamp,
        'version': 1
    }
    
    try:
        # Store event
        event_store_table.put_item(Item=event_data)
        
        # Publish event
        publish_event_to_bridge(event_data)
        
        return build_response(200, {
            'message': 'Order cancelled successfully',
            'reason': reason
        })
        
    except Exception as e:
        print(f"Error cancelling order: {str(e)}")
        return build_response(500, {'error': 'Failed to cancel order'})

def publish_event_to_bridge(event_data):
    try:
        # Convert Decimal values to float for JSON serialization
        event_data_for_bridge = convert_decimals_to_float(event_data['eventData'])
        
        print(f"Publishing event to EventBridge: {event_data['eventType']}")
        print(f"Event data: {event_data_for_bridge}")
        
        response = eventbridge.put_events(
            Entries=[
                {
                    'Source': 'fastfood.orders',
                    'DetailType': event_data['eventType'],
                    'Detail': json.dumps(event_data_for_bridge, default=str),
                    'EventBusName': 'FastFoodOrderBus'
                }
            ]
        )
        print(f"EventBridge response: {response}")
        
        # Check if there were any failures
        if response.get('FailedEntryCount', 0) > 0:
            print(f"Failed to publish some events: {response.get('Entries', [])}")
            raise Exception("Failed to publish event to EventBridge")
        
    except Exception as e:
        print(f"Error publishing event to EventBridge: {str(e)}")
        import traceback
        traceback.print_exc()
        raise

def convert_decimals_to_float(obj):
    """
    Recursively convert Decimal objects to float for JSON serialization
    """
    if isinstance(obj, list):
        return [convert_decimals_to_float(item) for item in obj]
    elif isinstance(obj, dict):
        return {key: convert_decimals_to_float(value) for key, value in obj.items()}
    elif isinstance(obj, Decimal):
        return float(obj)
    else:
        return obj

def build_response(status_code, body):
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type, Authorization'
        },
        'body': json.dumps(body, default=str)
    }
